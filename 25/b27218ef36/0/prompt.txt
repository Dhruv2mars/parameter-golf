   class BigramHash(nn.Module):
       """Hash-based n-gram lookup - captures local patterns cheaply"""
       def __init__(self, vocab_size, capacity=4096):
           super().__init__()
           self.table = nn.Embedding(vocab_size, 64)  # hash bucket features
           self.proj = nn.Linear(64, model_dim)

       def forward(self, x):
           # Simple: hash current + prev token
           prev = torch.roll(x, 1, dims=0)
           hash_idx = (x * 31 + prev) % self.capacity
           return self.proj(self.table(hash_idx))

   def compress_with_document_boundary(val_tokens, model, sp):
       """Split validation into documents and compress each separately"""
       # Find document boundaries (tokenizer has boundary markers)
       docs = split_by_boundary(val_tokens)

       total_bits = 0
       total_bytes = 0

       for doc in docs:
           # Model prediction
           with torch.no_grad():
               logits = model(doc[:-1])
               probs = F.softmax(logits, dim=-1)

           # Bits = -log2(p) for actual tokens
           bits = -torch.log2(probs[range(len(doc)-1), doc[1:]].clamp(min=1e-10))
           total_bits += bits.sum().item()

           # Bytes from tokenizer
           total_bytes += sum(len(sp.id_to_piece(t).encode('utf-8')) for t in doc)

       return total_bits / total_bytes
